		FormParameter fp = new FormParameter("桜木くん","1234");
		DataBaseObject db = new DataBaseObject();
		AccountDAO dao = new AccountDAO(db.getJDBC_URL(),db.getDB_USER(),db.getDB_PASS());
		UserAccount result = dao.findByUser(fp);

		if(result != null &&
